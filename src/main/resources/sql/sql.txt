CREATE TABLE TB_USER(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT,
	USER_KIND INT COMMENT '用户类型1：商家 2：买家',
	USER_NAME VARCHAR(10),
	USER_CODE VARCHAR(20),
	USER_PASS VARCHAR(20),
	USER_PHONE VARCHAR(20),
	LOCK_FLAG INT COMMENT '是否锁定：1为锁定，0不锁定',
	LAST_LOGIN_IP VARCHAR(30),
	LAST_LOGIN_DATE DATETIME,
	ACTIVE_FLAG INT COMMENT '数据是否有效：1为有效，0为无效',
	CREATE_BY VARCHAR(10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME
	) COMMENT '用户表'
	
CREATE TABLE TB_CATEGORY(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT,
	PAR_ID BIGINT COMMENT '父ID' COMMENT '-1：没有上级目录',
	LIST_NAME VARCHAR(20),
	ACTIVE_FLAG INT COMMENT '数据是否有效：1为有效，0为无效',
	CREATE_BY VARCHAR(10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME
) COMMENT '商品目录'

CREATE TABLE TB_PRODUCT(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT,
	PRODUCT_LIST_ID BIGINT,
	PRODUCT_NAME VARCHAR(20),
	BRAND_NAME VARCHAR(20),
	IMG VARCHAR(100),
	PRODUCT_PRICE DOUBLE,
	INVENTORY INT COMMENT '库存',
	GO_OUT_FLAG INT COMMENT'商品上下架信息 1：上架 0：下架',
	PRODUCT_INFO VARCHAR(100),
	ACTIVE_FLAG INT COMMENT '数据是否有效：1为有效，0为无效',
	CREATE_BY VARCHAR(10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME
) COMMENT'商品表'

CREATE TABLE TB_ORDER_LIST(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT,
	ORDER_ID BIGINT,
	PRODUCT_ID BIGINT,
	BUY_COUNT INT,
	SUM_PRICE DOUBLE,
	ACTIVE_FLAG INT COMMENT '数据是否有效：1为有效，0为无效',
	CREATE_BY VARCHAR(10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME
) COMMENT '订单列表'

CREATE TABLE TB_ORDER(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT,
	USER_ID BIGINT,
	ADDRESS_ID BIGINT,
	SUM_PRICE DOUBLE,
	ORDER_STATUS INT COMMENT '订单状态 1：待发货 2：已发货 3：已签收 0：已取消 ',
	PAY INT COMMENT '支付方式 1：微信 2：支付宝 3：银行卡',
	ACTIVE_FLAG INT COMMENT '数据是否有效：1为有效，0为无效',
	CREATE_BY VARCHAR(10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME
) COMMENT '订单'

CREATE TABLE TB_ADRESS(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT,
	USER_ID BIGINT,
	PROVINCE VARCHAR(10) COMMENT '省',
	CITY VARCHAR(10) COMMENT'市',
	DISTRICT VARCHAR(10) COMMENT'区',
	DETAILED_ADRESS VARCHAR(50) COMMENT'详细地址',
	ADRESS_PHONE VARCHAR(20) COMMENT'收货电话',
	POSTCODE VARCHAR(10) COMMENT '邮编',
	ACTIVE_FLAG INT COMMENT '数据是否有效：1为有效，0为无效',
	CREATE_BY VARCHAR(10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME
) COMMENT '地址'

CREATE TABLE TB_SHOPPING(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT,
	USER_ID BIGINT,
	PRODUCT_ID BIGINT,
	PAY_COUNT INT,
	SUM_PRICE DOUBLE,
	ACTIVE_FLAG INT COMMENT '数据是否有效：1为有效，0为无效',
	CREATE_BY VARCHAR(10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME
) COMMENT '购物车'
